import { MetadataRoute } from 'next'

export default function robots(): MetadataRoute.Robots {
  return {
    rules: {
      userAgent: '*',
      allow: '/',
      disallow: '/admin/', // Admin panel ko Google se chhupayein
    },
    sitemap: 'https://pak-vape-store-pods.vercel.app/sitemap.xml',
  }
}
